\defbeamertemplate*{title page}{customized}[1][]
{
\centering
  \usebeamerfont{title}\inserttitle\par
  \smallskip
  \usebeamerfont{subtitle}\usebeamercolor[fg]{subtitle}\insertsubtitle\par
  \bigskip
  \usebeamerfont{author}\usebeamercolor[fg]{author}\insertauthor\par
  \bigskip
  \includegraphics[height=3cm]{img/logo1000px.png}\\
  \bigskip
  \usebeamerfont{institute}\insertinstitute\par
  \smallskip
  \usebeamerfont{date}\insertdate\par
}

%%%%%%%%%%%%%%%%%% FOOTLINE %%%%%%%%%%%%%%%%%%%%%%%%
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% To insert the date in the footline use \insertdate
% To insert institue use \insertshortinstitute
\setbeamertemplate{footline}{
\hbox{%
   \begin{beamercolorbox}[wd = .3\paperwidth, ht=3ex, dp=2.5ex, center, leftskip=.4cm]{footlinecolorLeft}%
    \insertshortauthor
   \end{beamercolorbox}%
   \begin{beamercolorbox}[wd=.6\paperwidth, ht=3ex, dp=2.5ex, center, leftskip=.4cm,rightskip=.4cm]{footlinecolorCenter}%
	\insertsectionhead : \insertsubsectionhead
   \end{beamercolorbox}%
   \begin{beamercolorbox}[wd=.1\paperwidth, ht=3ex, dp=2.5ex, right, rightskip=.4cm]{footlinecolorRight}%
   \insertframenumber
   \end{beamercolorbox}
   }
}
